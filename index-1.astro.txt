<script>
  let activeCat = 'all';

  function filterGlossary() {
    const q = (document.getElementById('gloss-search') as HTMLInputElement).value.toLowerCase();
    const items = document.querySelectorAll('.glossary-item') as NodeListOf<HTMLElement>;
    let visible = 0;
    items.forEach(item => {
      const termMatch = item.dataset.term?.includes(q);
      const catMatch = activeCat === 'all' || item.dataset.cat === activeCat;
      const show = termMatch && catMatch;
      item.style.display = show ? '' : 'none';
      if (show) visible++;
    });
    (document.getElementById('gloss-empty') as HTMLElement).style.display = visible === 0 ? '' : 'none';
  }

  function filterCat(cat: string) {
    activeCat = cat;
    document.querySelectorAll('.cat-btn').forEach(btn => {
      const b = btn as HTMLElement;
      const isActive = b.dataset.cat === cat;
      b.className = isActive
        ? 'cat-btn active px-3 py-1 rounded-full text-xs border border-[#c9a227] text-[#c9a227] bg-[#c9a227]/10 transition-colors'
        : 'cat-btn px-3 py-1 rounded-full text-xs border border-[#222] text-[#555] hover:border-[#333] hover:text-[#888] transition-colors';
    });
    filterGlossary();
  }

  (window as any).filterGlossary = filterGlossary;
  (window as any).filterCat = filterCat;
</script>
